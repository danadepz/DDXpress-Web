-- DDXpress schema + seed (MySQL / MariaDB)
-- DB name: Ddxpress_db

CREATE DATABASE IF NOT EXISTS Ddxpress_db
  DEFAULT CHARACTER SET utf8mb4
  DEFAULT COLLATE utf8mb4_general_ci;

USE Ddxpress_db;

SET FOREIGN_KEY_CHECKS = 0;

DROP TABLE IF EXISTS parcel_status_history;
DROP TABLE IF EXISTS parcel;
DROP TABLE IF EXISTS payment;
DROP TABLE IF EXISTS booking;
DROP TABLE IF EXISTS service_type;
DROP TABLE IF EXISTS staff;
DROP TABLE IF EXISTS branch;
DROP TABLE IF EXISTS customer;

SET FOREIGN_KEY_CHECKS = 1;

-- 1) Customer
CREATE TABLE customer (
  id INT AUTO_INCREMENT PRIMARY KEY,
  full_name VARCHAR(120) NOT NULL,
  email VARCHAR(190) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  phone VARCHAR(40) NULL,
  address VARCHAR(255) NULL,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 2) Branch
CREATE TABLE branch (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(120) NOT NULL,
  address VARCHAR(255) NULL,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 3) Staff (includes admin + rider)
CREATE TABLE staff (
  id INT AUTO_INCREMENT PRIMARY KEY,
  full_name VARCHAR(120) NOT NULL,
  email VARCHAR(190) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  staff_role ENUM('staff','rider','admin') NOT NULL,
  branch_id INT NULL,
  is_active TINYINT(1) NOT NULL DEFAULT 1,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_staff_branch FOREIGN KEY (branch_id) REFERENCES branch(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 4) ServiceType
CREATE TABLE service_type (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(120) NOT NULL,
  base_fee DECIMAL(10,2) NOT NULL DEFAULT 0.00,
  per_kg_fee DECIMAL(10,2) NOT NULL DEFAULT 0.00,
  eta_days INT NOT NULL DEFAULT 2,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 5) Booking
CREATE TABLE booking (
  id INT AUTO_INCREMENT PRIMARY KEY,
  customer_id INT NOT NULL,
  service_type_id INT NOT NULL,
  origin_branch_id INT NOT NULL,
  destination_branch_id INT NOT NULL,
  sender_name VARCHAR(120) NOT NULL,
  sender_phone VARCHAR(40) NULL,
  sender_address VARCHAR(255) NULL,
  receiver_name VARCHAR(120) NOT NULL,
  receiver_phone VARCHAR(40) NULL,
  receiver_address VARCHAR(255) NULL,
  notes VARCHAR(255) NULL,
  status ENUM('CREATED','PAID','IN_TRANSIT','DELIVERED','CANCELLED') NOT NULL DEFAULT 'CREATED',
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_booking_customer FOREIGN KEY (customer_id) REFERENCES customer(id),
  CONSTRAINT fk_booking_service FOREIGN KEY (service_type_id) REFERENCES service_type(id),
  CONSTRAINT fk_booking_origin FOREIGN KEY (origin_branch_id) REFERENCES branch(id),
  CONSTRAINT fk_booking_dest FOREIGN KEY (destination_branch_id) REFERENCES branch(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 6) Payment
CREATE TABLE payment (
  id INT AUTO_INCREMENT PRIMARY KEY,
  booking_id INT NOT NULL,
  amount DECIMAL(10,2) NOT NULL,
  method ENUM('CASH','CARD','GCASH') NOT NULL DEFAULT 'CASH',
  status ENUM('PENDING','PAID','FAILED','REFUNDED') NOT NULL DEFAULT 'PENDING',
  paid_at DATETIME NULL,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_payment_booking FOREIGN KEY (booking_id) REFERENCES booking(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 7) Parcel
CREATE TABLE parcel (
  id INT AUTO_INCREMENT PRIMARY KEY,
  booking_id INT NOT NULL,
  tracking_number VARCHAR(32) NOT NULL UNIQUE,
  description VARCHAR(255) NOT NULL,
  declared_value DECIMAL(10,2) NOT NULL DEFAULT 0.00,
  weight_kg DECIMAL(10,2) NOT NULL DEFAULT 0.00,
  status ENUM(
    'PENDING_DROP_OFF',
    'PENDING_INSPECTION',
    'DECLINED',
    'READY_FOR_PICKUP',
    'OUT_FOR_DELIVERY',
    'IN_TRANSIT',
    'DELIVERED'
  ) NOT NULL DEFAULT 'PENDING_DROP_OFF',
  current_branch_id INT NULL,
  inspected_by_staff_id INT NULL,
  assigned_rider_id INT NULL,
  last_scan_at DATETIME NULL,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  CONSTRAINT fk_parcel_booking FOREIGN KEY (booking_id) REFERENCES booking(id),
  CONSTRAINT fk_parcel_branch FOREIGN KEY (current_branch_id) REFERENCES branch(id),
  CONSTRAINT fk_parcel_inspected_by FOREIGN KEY (inspected_by_staff_id) REFERENCES staff(id),
  CONSTRAINT fk_parcel_rider FOREIGN KEY (assigned_rider_id) REFERENCES staff(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 8) ParcelStatusHistory
CREATE TABLE parcel_status_history (
  id INT AUTO_INCREMENT PRIMARY KEY,
  parcel_id INT NOT NULL,
  status VARCHAR(40) NOT NULL,
  branch_id INT NULL,
  note VARCHAR(255) NULL,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_by_staff_id INT NULL,
  updated_by_rider_id INT NULL,
  CONSTRAINT fk_hist_parcel FOREIGN KEY (parcel_id) REFERENCES parcel(id),
  CONSTRAINT fk_hist_branch FOREIGN KEY (branch_id) REFERENCES branch(id),
  CONSTRAINT fk_hist_staff FOREIGN KEY (updated_by_staff_id) REFERENCES staff(id),
  CONSTRAINT fk_hist_rider FOREIGN KEY (updated_by_rider_id) REFERENCES staff(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
